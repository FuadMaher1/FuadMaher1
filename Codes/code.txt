  Create the security policy function
CREATE OR REPLACE FUNCTION get_user_dept_predicate (
    scheama_p IN VARCHAR2,
    table_p  IN VARCHAR2
) RETURN VARCHAR2 
IS
    v_user VARCHAR2(30);
BEGIN
    v_user := SYS_CONTEXT('USERENV', 'SESSION_USER');
    
    IF v_user = 'HR_MANAGER' THEN
        RETURN '1=1'; -- No restriction
    ELSIF v_user = 'SALES_USER' THEN
        RETURN 'DEPT_NAME = ''SALES''';
    ELSIF v_user = 'TECH_USER' THEN
        RETURN 'DEPT_NAME = ''TECH''';
    ELSE
        RETURN '1=2'; -- Deny access
    END IF;
END;
/
 

 -- Attach the policy using DBMS_RLS
BEGIN
    DBMS_RLS.ADD_POLICY (
        object_schema   => 'HR_SCHEAMA', 
        object_name     => 'EMPLOYEES_DATA',
        policy_name     => 'dept_security_policy',
        function_schema => 'HR_SCHEAMA',
        policy_function => 'get_user_dept_predicate',
        statement_types => 'SELECT'
    );
END;
/
